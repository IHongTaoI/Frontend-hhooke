# 🚗 SQL连接查询

SELECT * FROM order_items 
  LEFT OUTER JOIN orders 
  ON order_items.order_id = orders.id

SELECT * FROM `A` INNER JOIN `B`
┌────────┐
│ A  ┌───┼────┐
│    │ ∩ │    │
└────┼───┘  B │
     └────────┘
内部联接	∩
左外连接	A + ∩
右外连接	∩ + B
完全外部联接	A + ∩ + B
